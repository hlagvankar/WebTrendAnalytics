INSERT INTO TABLE FINAL_OUTPUT
SELECT DATE,CLICK_DAY,CLICK_HOUR,CLICK_WEEK,CLICK_COUNT,'','','SR_7'
FROM
(

SELECT TO_DATE(CLICK_TIME) DATE,DAY(CLICK_TIME) CLICK_DAY,
HOUR(CLICK_TIME) CLICK_HOUR,WEEKOFYEAR(CLICK_TIME) CLICK_WEEK,
COUNT(1) CLICK_COUNT
FROM
(

SELECT ip_add,MIN(click_time) CLICK_TIME,hit_bucket
FROM PARSE_2ND
WHERE URL LIKE '%<web-url>%'
AND REST_URL NOT LIKE '%Googlebot%'
AND TRIM
(
substring
(
regexp_replace(URL,'<web-url>',''),
0,
(INSTR(regexp_replace(URL,'<web-url>',''),'/'))-1
)
) <>''
GROUP BY ip_add,hit_bucket

) A GROUP BY 
TO_DATE(CLICK_TIME),DAY(CLICK_TIME),
HOUR(CLICK_TIME),WEEKOFYEAR(CLICK_TIME)
) A;
